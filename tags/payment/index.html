<?php
  /**
    * @desc Function to print variable for debugging
    */
    function printPre($data, $extended=false)
    {
        echo "<pre>";
        if($extended)
        {
            var_dump($data);
        }
        else
        {
            print_r($data);
        }
        echo "</pre>";
    }
?>
